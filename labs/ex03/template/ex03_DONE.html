def least_squares(y, tx):
    """calculate the least squares solution."""
    gram_matrix = tx.T.dot(tx)
    w = np.linalg.solve(gram_matrix, tx.T.dot(y))
    loss = compute_loss(y, tx, w)
    return loss, w

def build_poly(x, degree):
    """polynomial basis functions for input data x, for j=0 up to j=degree."""
    x = np.array(x)
    phi = np.array([x**i for i in range(degree+1)]).T
    return phi

def split_data(x, y, ratio, seed=1):
    """
    split the dataset based on the split ratio. If ratio is 0.8 
    you will have 80% of your data set dedicated to training 
    and the rest dedicated to testing
    Returns x_train, y_train, x_test, y_test
    """
    # Maximum number of elements to include in the train set
    max_ind = int(ratio * len(x))
    # set seed
    np.random.seed(seed)
    # Generate an ordered list of indices
    ids = np.arange(len(x))
    # Suffle indices
    np.random.shuffle(ids)
    #Get shuffled arrays
    x_shuf = x[ids]
    y_shuf = y[ids]
    # Return both subsets
    return x_shuf[:max_ind], y_shuf[:max_ind], x_shuf[max_ind:], y_shuf[max_ind:]

def ridge_regression(y, tx, lambda_):
    """implement ridge regression."""
    gram_matrix = tx.T.dot(tx)
    reg_term = 2*len(y)*lambda_*np.identity(gram_matrix.shape[0])
    w = np.linalg.solve(gram_matrix + reg_term, tx.T.dot(y))
    loss = compute_loss(y, tx, w)
    return loss, w

def ridge_regression_demo(x, y, degree, ratio, seed):
    """ridge regression demo."""
    # define parameter
    lambdas = np.logspace(-5, 0, 15)
    # split the data, and return train and test data
    x_train, y_train, x_test, y_test = split_data(x, y, ratio, seed)
    # form train and test data with polynomial basis function
    tx_train = build_poly(x_train, degree)
    tx_test = build_poly(x_test, degree)
    rmse_tr = []
    rmse_te = []
    for ind, lambda_ in enumerate(lambdas):
        # calcualte weight through least square.
        loss_train, w_train = ridge_regression(y_train, tx_train, lambda_)
        rmse_tr.append(np.sqrt(2*loss_train))
        rmse_te.append(np.sqrt(2*compute_loss(y_test, tx_test, w_train, kind = 'mse')))
        print("proportion={p}, degree={d}, lambda={l:.3f}, Training RMSE={tr:.3f}, Testing RMSE={te:.3f}".format(
               p=ratio, d=degree, l=lambda_, tr=rmse_tr[ind], te=rmse_te[ind]))
        
    # Plot the obtained results
    plot_train_test(rmse_tr, rmse_te, lambdas, degree)
